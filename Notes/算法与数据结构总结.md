#### 牛顿迭代法求平方根：

 https://www.cnblogs.com/upcan/p/9907402.html 

![1598092733476](images/算法与数据结构总结/1598092733476.png)

```c++
#include <iostream>
#include <cmath>
using namespace std;

// err 是允许的误差
const double err = 1e-8;

double NtSqrt(const double num)
{
    if (num < 0)
    {
        return -1;
    }
    else
    {
        double root = num;
        // 如果原值减去近似根的平方大于误差，继续循环
        while (abs(num - root * root) >= err)
        {
            // 得到下一个近似根
            root = (num / root + root) / 2.0;
        }
        return root;
    }
}

int main()
{
    double num;
    cout << "请输入一个数: ";
    cin >> num;
    double ans = NtSqrt(num);
    if (ans == -1)
    {
        cout << "负数没有平方根" << endl;
    }
    else
    {
        cout << num << " 的平方根是 " << ans << endl;
    }
    return 0;
}
```

```c++
vector<int> majorityElement(vector<int>& nums) 
{
    vector<int>ans;
    int n = nums.size();
    if(n<1)return ans;
    int cnt1=0,cnt2=0,tmp1=nums[0],tmp2=nums[0];
    for (int i = 0; i < n; i++) {
        if (nums[i] == tmp1)
            cnt1++;
        else if (nums[i] == tmp2)
            cnt2++;
        else if (cnt1 == 0) {
            tmp1 = nums[i];
            cnt1 = 1;
        } else if (cnt2 == 0) {
            tmp2 = nums[i];
            cnt2 = 1;
        } else {
            cnt1--;
            cnt2--;
        }
    }
    ans.push_back(tmp1);
    ans.push_back(tmp2);
    return ans;
}
```

```c++
// 查找是否存在
bool SearchMatrix(vector<vector<int> > matrix, int target) {
    if(matrix.size() == 0) return false;
    int m = matrix.size(), n = matrix[0].size();
    int mid, low = 0, high = n * m - 1;
    while(low <= high) {
        mid = (low + high) / 2;
        int r = mid / n;
        int c = mid % n;
        if(matrix[r][c] == target) {
            return true;
        }
        else if(matrix[r][c] < target) {
            low = mid + 1;
        } 
        else {
            high = mid - 1;
        }
    }
    return false;
}

// 查找次数
int SearchMatrixP(vector<vector<int> > matrix, int target) {
    if(matrix.size() == 0) return 0;
    int m = matrix.size(), n = matrix[0].size();
    int i = m - 1, j = 0, cnt = 0;
    while(i >= 0 && j < n) {
        if(target == matrix[i][j]) {
            cnt ++;
            j ++;
        }
        else if(target > matrix[i][j]) {
            j ++;
        }
        else {
            i --;
        }
    }
    return cnt;
}
```

#### 压缩字符串

```c++
class Solution {
public:
    int compress(vector<char>& chars) {
        int i;
        int len=0;
        
       for( i=0;i<chars.size();i++){
           int cnt=1;
            while(i+1<chars.size()&&chars[i]==chars[i+1]){
                cnt++;
                i++;
            }
           chars[len++]=chars[i];
           if(cnt==1)continue;
           for(char ch:to_string(cnt)){
               chars[len++]=ch;
           }
           
        }
        return len;
    }
};
```
